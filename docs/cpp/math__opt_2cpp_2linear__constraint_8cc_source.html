34BoundedLinearExpression LinearConstraint::AsBoundedLinearExpression() const {
35 LinearExpression terms;
36 for (const VariableId var : model_->variables_in_linear_constraint(id_)) {
37 terms +=
38 Variable(model_, var) * model_->linear_constraint_coefficient(id_, var);
39 }
40 return lower_bound() <= std::move(terms) <= upper_bound();
41}
